#include <Wire.h>
#include <RTClib.h>
#include <DHT.h>
#include <Adafruit_GFX.h>
#include <Adafruit_SSD1306.h>

#define SCREEN_WIDTH 128
#define SCREEN_HEIGHT 64

Adafruit_SSD1306 oled(SCREEN_WIDTH, SCREEN_HEIGHT, &Wire, -1);

#define DHTPIN 7
#define DHTTYPE DHT11

#define FAN_PIN 8
#define BULB2_PIN 9
#define BULB1_PIN 10
#define FLAME_PIN 11
#define LDR_PIN A0

DHT dht(DHTPIN, DHTTYPE);
RTC_DS1307 rtc;

void setup()
{
pinMode(FAN_PIN, OUTPUT);
pinMode(BULB1_PIN, OUTPUT);
pinMode(BULB2_PIN, OUTPUT);
pinMode(FLAME_PIN, INPUT);

Serial.begin(9600);

dht.begin();

if (!rtc.begin())
{
Serial.println("RTC Error");
}

oled.begin(SSD1306_SWITCHCAPVCC, 0x3C);
oled.clearDisplay();

Serial.println("System Started");
}

void loop()
{
float temp = dht.readTemperature();
float hum = dht.readHumidity();

int flame = digitalRead(FLAME_PIN);
int ldrValue = analogRead(LDR_PIN);

// LDR Control Bulb1
if (ldrValue < 300)
{
digitalWrite(BULB1_PIN, HIGH);
}
else
{
digitalWrite(BULB1_PIN, LOW);
}

// Flame Control Fan and Bulb2
if (flame == HIGH)
{
digitalWrite(FAN_PIN, HIGH);
digitalWrite(BULB2_PIN, HIGH);
}
else
{
digitalWrite(FAN_PIN, LOW);
digitalWrite(BULB2_PIN, LOW);
}

DateTime now = rtc.now();

// OLED DISPLAY
oled.clearDisplay();
oled.setTextSize(1);
oled.setTextColor(SSD1306_WHITE);

oled.setCursor(0,0);
oled.print("Time:");
oled.print(now.hour());
oled.print(":");
oled.print(now.minute());
oled.print(":");
oled.print(now.second());

oled.setCursor(0,12);
oled.print("Temp:");
oled.print(temp,0);
oled.print("C");

oled.setCursor(70,12);
oled.print("Hum:");
oled.print(hum,0);
oled.print("%");

oled.setCursor(0,24);
oled.print("LDR:");
oled.print(ldrValue);

oled.setCursor(0,36);
oled.print("Light:");
if(digitalRead(BULB1_PIN))
oled.print("ON");
else
oled.print("OFF");

oled.setCursor(0,48);
oled.print("Flame:");

if(flame == HIGH)
oled.print("DETECTED");
else
oled.print("NOT DET");

oled.setCursor(0,56);

if(flame == HIGH)
oled.print("Status: ALERT");
else
oled.print("Status: SAFE");

oled.display();

// Virtual Terminal Output
Serial.print("Time:");
Serial.print(now.hour());
Serial.print(":");
Serial.print(now.minute());
Serial.print(":");
Serial.print(now.second());

Serial.print(" Temp:");
Serial.print(temp);

Serial.print(" Hum:");
Serial.print(hum);

Serial.print(" LDR:");
Serial.print(ldrValue);

Serial.print(" Flame:");

if(flame == HIGH)
Serial.print("DETECTED");
else
Serial.print("NOT DETECTED");

Serial.print(" Status:");

if(flame == HIGH)
Serial.println("ALERT");
else
Serial.println("SAFE");

delay(1000);
}
